üíª JavaScript Integration Example

For more control, you can use JavaScript to dynamically create and manage the widget:

<!-- Button to trigger widget --> <button id="book-appointment">Book Appointment</button> <!-- Modal container --> <div id="widget-modal" style="display: none;"> <iframe id="appointment-widget" width="100%" height="700px" frameborder="0" ></iframe> </div> <script> // Configuration const config = { apiUrl: 'https://your-api.com/api', theme: 'light', primaryColor: '#0EA5E9' }; // Build widget URL const widgetUrl = `http://yoursite.com/widget?` + `apiUrl=${encodeURIComponent(config.apiUrl)}&` + `theme=${config.theme}&` + `primaryColor=${encodeURIComponent(config.primaryColor)}`; // Show widget on button click document.getElementById('book-appointment').addEventListener('click', function() { const modal = document.getElementById('widget-modal'); const iframe = document.getElementById('appointment-widget'); iframe.src = widgetUrl; modal.style.display = 'block'; }); // Listen for booking completion window.addEventListener('message', function(event) { if (event.data.type === 'APPOINTMENT_BOOKED') { console.log('Appointment booked:', event.data.data); // Close modal, show success message, etc. document.getElementById('widget-modal').style.display = 'none'; } }); </script>
